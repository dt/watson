@(g: game.Game) @main("Watson") {
@for(y <- 'a' to 'f') {
@for(x <- 1 to 6) {
@for(c <- 1 to 6) { @game.RowChoice(y-97, c-1).name }
}
}
@g.sameCol.map { c =>
@c.above.name ↓ @c.below.name
}
@g.adjCol.map { c =>
@c.x.name ↔ @c.y.name
} @g.leftToRight.map { c =>
@c.left.name → @c.right.name
}

How to Play

  • Tap to dismiss an option. Tap and hold to select
  • "x ↔ y" means "x appears in the column next to y"
  • "x → y" means "y appears in a column somewhere to the right of x"
  • "x ↓ y" means "x appears above y in the same column"

}
